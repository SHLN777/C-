全排列

代码整体思路如下：
perm({1,2,3})=1perm({2,3})+2perm({1,3})+3perm({1,2}) 
perm({2,3})=2perm({3})+3perm({2}) 
perm({1,3})=1perm({3})+3perm({1}) 
perm({1,2})=1perm({2})+2perm({1})

#include <iostream>
using namespace std;
void swap(int &a,int &b){
    int temp=a;
    a=b;
    b=temp;
}

void perm( int list[] , int low , int high ){
    if(low==high){   //当low==high时,此时list就是其中一个排列,输出list
        for(int i=0;i<=low;i++)
            cout<<list[i];
        cout<<endl;
    }
else{
        for(int i=low;i<=high;i++){//每个元素与第一个元素交换
            swap(list[i],list[low]);
            perm(list,low+1,high); //交换后,得到子序列,用函数perm得到子序列的全排列
            swap(list[i],list[low]);//最后,将元素交换回来,复原,然后交换另一个元素
        }
    }
}
int main()
{
int list[]={1,2,3};
perm(list,0,2);
return 0;
}

#include <iostream>
using namespace std;


template <class T>
void subset(T a[], int arry[], int index, int L)
{
    if(index == L)
    {
        cout<<"{";
        for(int i=0; i<L; i++){
        
            if(arry[i]==1)
                cout<<a[i];
        }
        cout<<"}"<<endl;
    
    }
    else{
    
        arry[index]=0;
        subset(a,arry, index+1, L);
        arry[index]=1;
        subset(a,arry, index+1, L);


    }
}



int main()
{
    char ch[]={'a','b','c','d','e'};
    int arry[5];
    subset<char>(ch, arry, 0, 5);


    return 0;
}
