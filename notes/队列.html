循环队列
队头、队尾、队大小、动态分配内存
front指向对头元素的上一个，rear指向队尾元素
(rear-front+size)%size

class SeqQueue
{
private:
    int * base;
    int front;
    int rear;
    int size;
public:
    SeqQueue(int Queuesize=100)
    {
        base = new int[Queuesize];
        front = 0;
        rear = 0;
        size = Queuesize;
    }
    ~SeqQueue()
    {
        delete[] base;
    }
    int Empty_Queue()
    {
        return (front != rear)
    }
    int En_Queue(int e)
    {
        if( (rear+1)%size==front )
        {
            rear = (rear+1)%size;
            base[rear] = e;
            return 1;
        }
        else
            return 0;
    }
    int De_Queue(int &e)
    {
        if(front != rear)
        {
            front = (front+1)%size;
            e = base[front];
            return 1;
        }
        else
            return 0;
    }
    int Front_Queue(int &e)
    {
        if(front != rear)
        {   e = base[(front+1)%size];
            return 1;
        }
        else
            return 0;
    }
};


链队列
front指向第一个节点、rear指向最后一个节点

class LinkQueue
{
private:
    QueueNode *front;
    QueueNode *rear;
public:
    LinkQueue()
    {
        front = NULL;
        rear = NULL;
    }
    ~LinkQueue()
    {
        QueueNode *p,*q;
        p = front;
        while(p)
        {
            q=p;
            p=p->next;
            delete q;
        }
        front = NULL;
        rear = NULL;
    }
    int Empty_Queue()
    {
        return (front==NULL&&rear==NULL);
    }
    int En_Queue(int e)
    {
        QueueNode *p=new QueueNode;
        if(p)  //申请节点成功
        {
            p->data=e;
            if(rear)  //非空
            {
                rear->next=p;
            }
            else  //空
            {
                front=rear=p;
                return 1;
            }
        }
        else
            return 0;
    }
    int De_Queue(int &e)
    {    
        QueueNode *p;
        if(!Empty_Queue)
        {
            p = front;
            e = p->data;
            front = front->next;
            if(!front)
            {
                rear = NULL;
                delete p;
            }
        }
    int Front_Queue(int &e)
    {
        if(!Empty_Queue())
        {
            e=front->data;
            return 1;
        }
        else
            return 0;
    }

    }
