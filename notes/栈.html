#include <iostream>
using namespace std;

//顺序栈
class Stack
{
public:
    Stack(int sizestack);
    ~Stack();
    int Empty_Stack()
    {
        if ((top <= base))
        {
            return 1;
        }
        else
        {
            return 0;
        }
    }
    int push_stack(int elem)
    {
        if (top - base == size)
        {
            return 0;
        }
        else
        {
            *top = elem;
            top++;
            return 1;
        }
    }
    int pop_stach(int &elem)
    {
        if(top>base)
        {
            --top;
            elem = *top;
            return 1;
        }
        else
        {
            return 0;
        }
    }
    int get_stacktop(int &elem)
    {
        if (top > base)
        {
            return *(top-1);
        }
    }

private:
    int *top;
    int *base;
    int size;
};


Stack::Stack(int stacksize)
{
    base = new int[stacksize];
    top = base;
    size = stacksize;
}


Stack::~Stack()
{
    delete[] base;
    base = NULL;
    top = NULL;
}

int main()
{
    int a;
    int b;
    Stack s(10);
    for(int i = 0; i < 10; i++)
    {
        s.push_stack(i);
    }
    s.Empty_Stack();
    s.get_stacktop(a);
    s.pop_stach(b);
    cout << a << b << endl;
}
