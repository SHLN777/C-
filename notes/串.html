构造函数的实现

String::String()
{
    size=1;
    str=new char[size];
    if(str == NULL)
    {
        exit(1);
    }
    str[0]='\0';
}

String::String(const char *s)
{
    size = strlen(s)+1;
    str=new char[size];
    if(str==NULL)
        exit(1);
    strcpy(str,s);
}

String::String(const String &obj)
{
    size=obj.size;
    str=new char[size+1];
    if(str==NULL)
        exit(1);
    strcpy(str,obj,str);
}
